Fly

Fly 飞行动画

js调用

基本形式

Destination
						

						$("#btn-fly").on("click",function(){
							$(this).fly(
								      {
								            start:{
								              left:$("#btn-fly").offset().left,
								              top:$("#btn-fly").offset().top
								             },
								             end:{
								              left:$("#btn-end").offset().left,
								              top:$("#btn-end").offset().top
								             }
								      }		
							);
						});

指定飞行元素

target
						

						$("#btn-fly1").on("click",function(){
							$(this).fly(
								      {
								            start:{
								              left:$("#btn-fly1").offset().left,
								              top:$("#btn-fly1").offset().top
								             },
								             end:{
								              left:$("#btn-end1").offset().left,
								              top:$("#btn-end1").offset().top
								             }
								             targetid:""
								      }		
							);
						});

调用方式

通过方法$(selector).fly(options)进行调用

opstions参数描述如下：

  • start：起始坐标对象，包括left和top两个坐标属性，用数值表示。
  • end：终点坐标对象，包含left和top两个坐标属性，用数值表示。
  • target：目标元素的id，需要进行飞行的元素的id，如果为空，则默认为selector选中的元素。
  • class：坐标飞行动画，可以为空，默认为一个缩放过程。
  • callback：回调函数，运行完之后调用的函数。
						
					
					var options ={
						            start:             //起始坐标，可以获取，也可以直接用数字坐标表示
						            {
						              left:$("#start").offset().left,   
						              top:$("#start").offset().top
						             },
						             end:              //终点坐标，可以获取目标的坐标，也可以根据需要用数字坐标表示
						             {
						              left:$("#btn-end").offset().left,
						              top:$("#btn-end").offset().top
						             },
						             class:"className" , //飞行过程动画，用类命名，可以添加自定义的类，默认为"fly"，可不填
						             callback:callbackfunc,  //回调函数
						             target: id        //需要进行飞行的对象的id，默认为jQuery对象本身
						          }
						          
					 //带jQuery对象的调用,target可以声明，也可以不声明
				     $().fly(options);
				     
				     //无对象的调用，target需要自主声明
				     $.fly(options);
